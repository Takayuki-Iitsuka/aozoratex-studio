        % --- 1. 四隅の幾何学模様 ---
        % 左上

                         ($(current page.north west) + (\margin, -\margin)$) -- ++(0, -15mm);
        \draw[cornerIn]  ($(current page.north west) + (\margin+2mm, -\margin-2mm)$) -- ++(8mm, 0)

        % 右上
        \draw[cornerOut] ($(current page.north east) + (-\margin, -\margin)$) -- ++(-15mm, 0)
                         ($(current page.north east) + (-\margin, -\margin)$) -- ++(0, -15mm);
        \draw[cornerIn]  ($(current page.north east) + (-\margin-2mm, -\margin-2mm)$) -- ++(-8mm, 0)
                         ($(current page.north east) + (-\margin-2mm, -\margin-2mm)$) -- ++(0, -8mm);
        % 左下

                         ($(current page.south west) + (\margin, \margin)$) -- ++(0, 15mm);
        \draw[cornerIn]  ($(current page.south west) + (\margin+2mm, \margin+2mm)$) -- ++(8mm, 0)
                         ($(current page.south west) + (\margin+2mm, \margin+2mm)$) -- ++(0, 8mm);
        % 右下
        \draw[cornerOut] ($(current page.south east) + (-\margin, \margin)$) -- ++(-15mm, 0)
                         ($(current page.south east) + (-\margin, \margin)$) -- ++(0, 15mm);
        \draw[cornerIn]  ($(current page.south east) + (-\margin-2mm, \margin+2mm)$) -- ++(-8mm, 0)
                         ($(current page.south east) + (-\margin-2mm, \margin+2mm)$) -- ++(0, 8mm);

        % --- 2. 上部（ヘッダー領域） ---
        % 左上ギリギリに本の名前
        \node[anchor=north west, font=\small\sffamily, text=darkgray]

% ダミーテキスト（青空文庫風）
「ではみなさんは、そういうふうに川だと云われたり、乳の流れたあとだと云われたりしていたこのぼんやりと白いものがほんとうは何かご承知ですか。」先生は、黒板に吊した大きな黒い星座の図の、上から下へ白くけぶった銀河帯のようなところを指しながら、みんなに問をかけました。

カムパネルラが手をあげました。それから四五人手をあげました。ジョバンニも手をあげようとして、急いでそのままやめました。たしかにあれがみんな星だと、いつか雑誌で読んだのでしたが、このごろはジョバンニはまるで毎日教室でもねむく、本を読むひまも読む本もないので、なんだかどんなこともよくわからないという気がするのでした。

\newpage

ところが先生は早くもそれを見附けたのでした。
「ジョバンニさん。あなたはわかっているのでしょう。」
ジョバンニは勢いよく立ちあがりましたが、立って見るともうはっきりとそれを答えることができないのでした。ザネリが前の席からふりかえって、ジョバンニを見てくすっとわらいました。ジョバンニはもうどぎまぎしてまっ赤になってしまいました。先生がまた云いました。
「大きな望遠鏡で銀河をよっく調べると銀河は大体何でしょう。」
