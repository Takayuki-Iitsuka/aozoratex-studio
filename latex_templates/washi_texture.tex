% washi texture snippet (overlay-ready)
% section={{section_label}}

\definecolor{washiCloudA}{HTML}{{{cloud_color_a}}}
\definecolor{washiCloudB}{HTML}{{{cloud_color_b}}}
\definecolor{washiCloudC}{HTML}{{{cloud_color_c}}}
\definecolor{washiFiberA}{HTML}{{{fiber_color_a}}}
\definecolor{washiFiberB}{HTML}{{{fiber_color_b}}}
\definecolor{washiFiberC}{HTML}{{{fiber_color_c}}}
\definecolor{washiSpeckA}{HTML}{{{speck_color_a}}}
\definecolor{washiSpeckB}{HTML}{{{speck_color_b}}}

\fill[fill={rgb,255:red,248;green,241;blue,222},opacity={{base_fill_opacity}}]
  (current page.south west) rectangle (current page.north east);

% subtle laid lines (suno-me)
\foreach \x in {0,{{laid_step_x}},...,{{laid_x_max}}} {
  \draw[line width=0.09pt, opacity={{line_x_opacity}}]
    ($(current page.south west)+(\x mm,0)$) -- ($(current page.north west)+(\x mm,0)$);
}
\foreach \y in {0,{{laid_step_y}},...,{{laid_y_max}}} {
  \draw[line width=0.08pt, opacity={{line_y_opacity}}]
    ($(current page.south west)+(0,\y mm)$) -- ($(current page.south east)+(0,\y mm)$);
}

\directlua{
  if not aozora_washi_seeded then
    local seed = os.time() + math.floor(1000000 * os.clock())
    math.randomseed(seed)
    aozora_washi_seeded = true
  end

  local function randf(a, b)
    return a + (b - a) * math.random()
  end

  local x_extent = {{x_extent_mm}}
  local y_extent = {{y_extent_mm}}

  local cloud_colors = {"washiCloudA", "washiCloudB", "washiCloudC"}
  local fiber_colors = {"washiFiberA", "washiFiberB", "washiFiberC"}
  local speck_colors = {"washiSpeckA", "washiSpeckB"}

  -- pulp clouds (color variation)
  for i = 1, {{patch_count}} do
    local x = randf(-x_extent, x_extent)
    local y = randf(-y_extent, y_extent)
    local rx = randf({{patch_rx_min_mm}}, {{patch_rx_max_mm}})
    local ry = randf({{patch_ry_min_mm}}, {{patch_ry_max_mm}})
    local rot = randf(0.0, 180.0)
    local op = randf({{patch_op_min}}, {{patch_op_max}})
    local c = cloud_colors[math.random(#cloud_colors)]
    tex.sprint(string.format(
      "\\fill[%s,opacity=%.4f] ($(current page.center)+(%.2fmm,%.2fmm)$) ellipse [x radius=%.2fmm,y radius=%.2fmm,rotate=%.1f];",
      c, op, x, y, rx, ry, rot
    ))
  end

  -- long fibers (random curves)
  for i = 1, {{fiber_count}} do
    local x1 = randf(-x_extent, x_extent)
    local y1 = randf(-y_extent, y_extent)
    local len = randf({{fiber_len_min_mm}}, {{fiber_len_max_mm}})
    local ang = randf(0.0, 2.0 * math.pi)
    local bend = randf(-{{fiber_bend_max_mm}}, {{fiber_bend_max_mm}})
    local x2 = x1 + len * math.cos(ang)
    local y2 = y1 + len * math.sin(ang)
    local cx = (x1 + x2) * 0.5 + bend
    local cy = (y1 + y2) * 0.5 - bend
    local op = randf({{fiber_op_min}}, {{fiber_op_max}})
    local lw = randf({{fiber_lw_min_mm}}, {{fiber_lw_max_mm}})
    local c = fiber_colors[math.random(#fiber_colors)]
    tex.sprint(string.format(
      "\\draw[%s,opacity=%.4f,line width=%.3fmm,line cap=round] ($(current page.center)+(%.2fmm,%.2fmm)$) .. controls ($(current page.center)+(%.2fmm,%.2fmm)$) .. ($(current page.center)+(%.2fmm,%.2fmm)$);",
      c, op, lw, x1, y1, cx, cy, x2, y2
    ))
  end

  -- specks / bark dust
  for i = 1, {{speck_count}} do
    local x = randf(-x_extent, x_extent)
    local y = randf(-y_extent, y_extent)
    local r = randf({{speck_radius_min_mm}}, {{speck_radius_max_mm}})
    local op = randf({{speck_op_min}}, {{speck_op_max}})
    local c = speck_colors[math.random(#speck_colors)]
    tex.sprint(string.format(
      "\\fill[%s,opacity=%.4f] ($(current page.center)+(%.2fmm,%.2fmm)$) circle (%.3fmm);",
      c, op, x, y, r
    ))
  end
}
